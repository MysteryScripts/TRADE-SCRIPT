local function showNotification(text, enabled)
    local emoji = enabled and "" or ""
    notificationText.Text = text .. " has been " .. (enabled and "enabled" or "disabled") .. " " .. emoji
    notificationFrame.Visible = true
    notificationFrame.BackgroundTransparency = 0.5
    wait(2)
    for i = 0.5, 1, 0.05 do
        notificationFrame.BackgroundTransparency = i
        wait(0.05)
    end
    notificationFrame.Visible = false
end
